- [ ] **Step 5: Update orchestrator dispatch in `src/plan/orchestrator.ts`**

```ts
async function bikeRouteCall(
  s: SearchState, from: LatLon, to: LatLon,

    const { DAY_RIDE_CUSTOM_MODEL } = await import('./types');

```

Replace with:

```ts
async function bikeRouteCall(
  s: SearchState, from: LatLon, to: LatLon,
): Promise<RouteResult> {
  if (s.req.goal === 'day-ride' || s.req.goal === 'max-path') {
    const { DAY_RIDE_CUSTOM_MODEL, MAX_PATH_CUSTOM_MODEL } = await import('./types');
    const model = s.req.goal === 'max-path' ? MAX_PATH_CUSTOM_MODEL : DAY_RIDE_CUSTOM_MODEL;
    const r = await s.deps.external.ghRouteCustom(from, to, model);
    if (r) return { km: r.km, min: r.min, geometry: r.geometry };
  }
  return s.deps.external.osrmRoute('bicycle', from, to);
}
```

```ts
  it('--goal max-path Hurstbridge → Darebin reaches >=95% on dedicated bike paths', async () => {
    const result = await plan({
      from: { lat: -37.6394, lon: 145.192017 },   // Hurstbridge
      to:   { lat: -37.7749634, lon: 145.038483 }, // Darebin
      departUtc: undefined,
      arriveByUtc: undefined,
      minBikeKm: 0, maxBikeKm: 60,
      maxTransfers: 0, enrich: true,
      preferBikePath: false, goal: 'max-path',
      mode: 'bike-only', hillWeight: 0,
    });
    expect(result.itineraries).toHaveLength(1);
    const it = result.itineraries[0];
    expect(it.bikeKm).toBeGreaterThan(30);  // longer than direct
    if (typeof it.bikeKmOnPath === 'number' && it.bikeKm > 0) {
      const pct = it.bikeKmOnPath / it.bikeKm;
      expect(pct).toBeGreaterThanOrEqual(0.95);  // 95%+ on dedicated paths
    }
  }, 60_000);
```
