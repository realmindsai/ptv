# ptv-chat Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Standalone web app where a user chats with Claude, which uses a tool loop (geocode / plan / bike_route / stops) to produce multiple selectable itineraries rendered as color-coded polylines on a Leaflet map. Deploys as its own Docker container on totoro, tailnet-only, authenticated against the developer's Claude Code subscription.

**Architecture:** New Fastify app at `src/chat/` sharing existing PTV/orchestrator/external code via imports. Each `POST /api/chat` opens an SSE stream for a single user turn driven by the Anthropic Claude Agent SDK in-process. Tool wrappers do double-duty: they side-channel full itinerary geometry to the browser as `path_add` events while returning compact summaries to Claude (token discipline). The browser is a vanilla-TS SPA (Leaflet + esbuild bundle) with a chat dock, a tool-call log overlay, and a Leaflet map showing all candidates with click-to-activate.

**Tech Stack:** TypeScript, Fastify 5, `@anthropic-ai/claude-agent-sdk`, Zod, Leaflet, vitest, Playwright, esbuild, Docker.

**Spec:** `docs/superpowers/specs/2026-05-19-ptv-chat-design.md`

```ts
import { z } from 'zod';
import type { Nominatim } from '../../server/nominatim';
import type { ChatCtx } from '../types';

const zArgs = z.object({ query: z.string().min(1) });
type Args = z.infer<typeof zArgs>;

export function makeGeocodeTool(ctx: ChatCtx, nominatim: Nominatim) {
  return {
    name: 'geocode' as const,
    description: 'Resolve a Melbourne-biased place name or address to {lat, lon}.',
    schema: zArgs,
    handler: async (args: Args) => {
      const hits = await nominatim.search(args.query);
      const first = hits[0];
      if (!first) return { ok: false as const, error: `No match for "${args.query}"` };
      return {
        ok: true as const,
        lat: first.lat,
        lon: first.lon,
        displayName: first.display_name,
      };
    },
  };
}
```

```ts
import { z } from 'zod';
import type { ChatCtx } from '../types';
import type { Itinerary, PlanRequest } from '../../plan/types';

const zLatLon = z.object({ lat: z.number(), lon: z.number() });

export const zPlanArgs = z.object({
  from: zLatLon,
  to: zLatLon,
  depart: z.string().optional(),
  arriveBy: z.string().optional(),
  mode: z.enum(['bike-train', 'bike-only']).default('bike-train'),
  goal: z.enum(['commute', 'day-ride', 'max-path']).default('commute'),
  maxTransfers: z.number().int().min(0).max(2).default(1),
  preferBikePath: z.boolean().optional(),
  hillWeight: z.number().optional(),
});
export type PlanArgs = z.infer<typeof zPlanArgs>;

export type PlanFn = (req: PlanRequest) => Promise<Itinerary[]>;

const PALETTE = ['#e6194b', '#3cb44b', '#4363d8', '#f58231', '#911eb4', '#008080'];
let paletteCursor = 0;
function nextColor(): string {
  const c = PALETTE[paletteCursor % PALETTE.length];
  paletteCursor++;
  return c;
}

export function makePlanTool(
  ctx: ChatCtx,
  planFn: PlanFn,
  genId: () => string = () => `p-${Math.random().toString(36).slice(2, 10)}`,
) {
  return {
    name: 'plan' as const,
    description:
      'Plan a bike+train (or bike-only) trip between two coordinates in Melbourne. ' +
      'Returns a count + per-itinerary summary. The full geometry is sent to the user ' +
      'map directly. Call multiple times to compare goals (commute vs day-ride).',
    schema: zPlanArgs,
    handler: async (args: PlanArgs) => {
      const req: PlanRequest = {
        from: args.from,
        to: args.to,
        depart: args.depart,
        arriveBy: args.arriveBy,
        mode: args.mode,
        goal: args.goal,
        maxTransfers: args.maxTransfers,
        preferBikePath: args.preferBikePath,
        hillWeight: args.hillWeight,
        enrich: true,
      };
      const itins = await planFn(req);
      if (itins.length === 0) {
        return { ok: false as const, error: 'No itineraries found' };
      }
      const summaries = itins.map((it) => {
        const id = genId();
        ctx.emit({
          type: 'path_add',
          pathId: id,
          label: it.label ?? 'unlabeled',
          color: nextColor(),
          itinerary: it,
        });
        return {
          label: it.label ?? 'unlabeled',
          distanceKm: it.distanceKm,
          durationMin: it.durationMin,
          transfers: it.transfers ?? 0,
          bikeKmOnPath: it.bikeKmOnPath,
        };
      });
      return { ok: true as const, itineraryCount: itins.length, summaries };
    },
  };
}
```

> **Note:** This file imports `PlanRequest` and `Itinerary` from `src/plan/types.ts`. If the existing `PlanRequest` differs in field names, adjust the request-construction block above. Do not modify `src/plan/types.ts`.

```ts
import { z } from 'zod';
import type { ChatCtx } from '../types';

const zLatLon = z.object({ lat: z.number(), lon: z.number() });

export const zBikeRouteArgs = z.object({
  from: zLatLon,
  to: zLatLon,
  goal: z.enum(['commute', 'day-ride', 'max-path']).default('commute'),
});
export type BikeRouteArgs = z.infer<typeof zBikeRouteArgs>;

export type GhRouteFn = (args: BikeRouteArgs) => Promise<{
  distanceKm: number;
  durationMin: number;
  ascendM: number;
  descendM: number;
  bikeKmOnPath: number;
  polyline: Array<[number, number]>;
}>;

const PALETTE = ['#e6194b', '#3cb44b', '#4363d8', '#f58231', '#911eb4', '#008080'];
let cursor = 0;
const nextColor = () => PALETTE[cursor++ % PALETTE.length];

export function makeBikeRouteTool(
  ctx: ChatCtx,
  ghRouteFn: GhRouteFn,
  genId: () => string = () => `b-${Math.random().toString(36).slice(2, 10)}`,
) {
  return {
    name: 'bike_route' as const,
    description:
      'Pure bicycle routing between two coords. goal=commute (fastest/safest), ' +
      'day-ride (prefers cycleways), or max-path (longest on dedicated path).',
    schema: zBikeRouteArgs,
    handler: async (args: BikeRouteArgs) => {
      const r = await ghRouteFn(args);
      const id = genId();
      ctx.emit({
        type: 'path_add',
        pathId: id,
        label: `bike (${args.goal})`,
        color: nextColor(),
        itinerary: {
          label: `bike-${args.goal}`,
          distanceKm: r.distanceKm,
          durationMin: r.durationMin,
```

```ts
import { query, tool } from '@anthropic-ai/claude-agent-sdk';
import type { ChatRequest, ChatCtx, SseEvent } from './types';

export type ToolBundle = {
  geocode: any;
  plan: any;
  bike_route: any;
  search_stops: any;
  nearby_stops: any;
};

export type RunTurnOpts = {
  tools: ToolBundle;
  model?: string;
  maxToolUses?: number;
};

const SYSTEM_PROMPT = (origin?: { lat: number; lon: number }, today = new Date()) => `
You help plan bike + train trips in Melbourne, Australia.

Available tools:
- geocode: place name -> {lat, lon}
- plan: bike+train (or bike-only) trip between two coords, with goal=commute|day-ride|max-path
- bike_route: pure bicycle routing
- search_stops: find PTV stops by name
- nearby_stops: find PTV stops near a coord

Workflow:
1. Geocode any place names not already coordinates.
2. Call plan (or bike_route for pure-bike asks). You may call it multiple times to
   compare goals or modes; each call adds candidate path(s) to the user's map.
3. Reply concisely. Name each path. Don't repeat geometry — the user sees polylines.

Today is ${today.toISOString().slice(0, 10)}.
Origin hint: ${origin ? `${origin.lat},${origin.lon}` : 'unknown'}.
`.trim();

export async function* runTurn(
  req: ChatRequest,
  opts: RunTurnOpts,
): AsyncGenerator<SseEvent> {
  yield { type: 'turn_start' };
  try {
    const sdkTools = [
      tool('geocode',      opts.tools.geocode.description,      opts.tools.geocode.schema,      opts.tools.geocode.handler),
      tool('plan',         opts.tools.plan.description,         opts.tools.plan.schema,         opts.tools.plan.handler),
      tool('bike_route',   opts.tools.bike_route.description,   opts.tools.bike_route.schema,   opts.tools.bike_route.handler),
      tool('search_stops', opts.tools.search_stops.description, opts.tools.search_stops.schema, opts.tools.search_stops.handler),
      tool('nearby_stops', opts.tools.nearby_stops.description, opts.tools.nearby_stops.schema, opts.tools.nearby_stops.handler),
    ];
    const stream = query({
      prompt: req.messages,
      systemPrompt: SYSTEM_PROMPT(req.origin),
      model: req.model ?? opts.model ?? process.env.MODEL ?? 'claude-sonnet-4-6',
      tools: sdkTools,
      maxToolUses: opts.maxToolUses ?? 10,
    });
    for await (const ev of stream as AsyncIterable<any>) {
      const mapped = mapSdkEvent(ev);
      if (mapped) yield mapped;
    }
  } catch (err: any) {
    yield { type: 'error', message: err?.message ?? String(err) };
  } finally {
    yield { type: 'turn_end' };
  }
}

export function mapSdkEvent(ev: any): SseEvent | null {
  if (!ev || typeof ev !== 'object') return null;
  switch (ev.type) {
    case 'content_block_delta':
      if (ev.delta?.type === 'text_delta') {
        return { type: 'text_delta', delta: ev.delta.text };
      }
      return null;
    case 'tool_use_start':
      return { type: 'tool_call', id: ev.tool_use_id, name: ev.name, args: ev.input };
    case 'tool_result': {
      const ok = ev.content?.ok !== false;
      const summary = typeof ev.content === 'object' && ev.content !== null
        ? JSON.stringify(ev.content).slice(0, 500)
        : String(ev.content ?? '');
      return { type: 'tool_result', id: ev.tool_use_id, ok, summary };
    }
    default:
      return null;
  }
}
```

> **Note:** Agent SDK event names above (`content_block_delta`, `tool_use_start`, `tool_result`, `message_stop`) are best-effort guesses from the SDK's public shape. When implementing, log raw `for await (const ev of query(...))` output once to confirm — then adjust `mapSdkEvent` keys to match. If the SDK exposes its own typed event union, import and use that instead of `any`. Adjust the test scripted events to match real event shapes.

```ts
import type { FastifyInstance } from 'fastify';
import type { ChatRequest, SseEvent } from '../types';
import { encodeSseEvent } from '../sse';

export type RunTurnFn = (req: ChatRequest, opts: any) => AsyncGenerator<SseEvent>;

export function registerChat(
  app: FastifyInstance,
  deps: { runTurnFn: RunTurnFn; buildTools: (ctx: any) => any },
): void {
  app.post('/api/chat', async (req, reply) => {
    const body = req.body as ChatRequest;
    reply.raw.writeHead(200, {
      'content-type': 'text/event-stream',
      'cache-control': 'no-cache, no-transform',
      'connection': 'keep-alive',
      'x-accel-buffering': 'no',
    });
    const emit = (ev: SseEvent) => {
      reply.raw.write(encodeSseEvent(ev));
    };
    const tools = deps.buildTools({ emit, origin: body.origin });
    try {
      for await (const ev of deps.runTurnFn(body, { tools })) {
        reply.raw.write(encodeSseEvent(ev));
      }
    } catch (err: any) {
      reply.raw.write(encodeSseEvent({ type: 'error', message: err?.message ?? 'unknown' }));
      reply.raw.write(encodeSseEvent({ type: 'turn_end' }));
    } finally {
      reply.raw.end();
    }
    return reply;
  });
}
```

```ts
import type { State, Action, Message, Path } from './types';

export function initialState(): State {
  return {
    messages: [],
    currentTurnPaths: [],
    activePathId: null,
    logEntries: [],
    logOpen: false,
    dockCollapsed: false,
    streaming: false,
    assistantBuffer: '',
  };
}

export function reduce(state: State, action: Action): State {
  switch (action.type) {
    case 'user_send':
      return {
        ...state,
        messages: [...state.messages, { role: 'user', content: action.content }],
        currentTurnPaths: [],
        logEntries: [],
        assistantBuffer: '',
        activePathId: null,
        streaming: true,
      };
    case 'turn_start':
      return { ...state, streaming: true };
    case 'text_delta':
      return { ...state, assistantBuffer: state.assistantBuffer + action.delta };
    case 'tool_call':
      return { ...state, logEntries: [...state.logEntries, action.entry] };
    case 'tool_result':
      return {
        ...state,
        logEntries: state.logEntries.map(e =>
          e.id === action.id ? { ...e, result: action.result, finishedAt: Date.now() } : e
        ),
      };
    case 'path_add':
      return { ...state, currentTurnPaths: [...state.currentTurnPaths, action.path] };
    case 'set_active':
      return {
        ...state,
        activePathId: state.activePathId === action.id ? null : action.id,
      };
    case 'turn_end': {
      const messages = state.assistantBuffer
        ? [...state.messages, { role: 'assistant', content: state.assistantBuffer } as Message]
        : state.messages;
      return { ...state, messages, assistantBuffer: '', streaming: false };
    }
    case 'toggle_log':
      return { ...state, logOpen: !state.logOpen };
    case 'toggle_dock':
      return { ...state, dockCollapsed: !state.dockCollapsed };
    case 'reset_chat':
      return initialState();
    case 'error':
      return {
        ...state,
        streaming: false,
        messages: [...state.messages, { role: 'assistant', content: `⚠ ${action.message}` }],
      };
  }
}
```

```ts
import type { SseEvent } from './types';

export function parseSseChunks(buf: string): { events: SseEvent[]; rest: string } {
  const events: SseEvent[] = [];
  let i = 0;
  while (true) {
    const sep = buf.indexOf('\n\n', i);
    if (sep === -1) break;
    const chunk = buf.slice(i, sep);
    if (chunk.startsWith('data: ')) {
      try { events.push(JSON.parse(chunk.slice('data: '.length))); }
      catch { /* skip malformed */ }
    }
    i = sep + 2;
  }
  return { events, rest: buf.slice(i) };
}

export async function streamChat(
  body: unknown,
  onEvent: (ev: SseEvent) => void,
  signal?: AbortSignal,
): Promise<void> {
  const res = await fetch('/api/chat', {
    method: 'POST',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify(body),
    signal,
  });
  if (!res.body) throw new Error('No response body');
  const reader = res.body.getReader();
  const decoder = new TextDecoder();
  let buf = '';
  while (true) {
    const { value, done } = await reader.read();
    if (done) break;
    buf += decoder.decode(value, { stream: true });
    const { events, rest } = parseSseChunks(buf);
    for (const ev of events) onEvent(ev);
    buf = rest;
  }
}
```

```ts
import type { Itinerary, Path } from './types';

// Leaflet is loaded from /static/leaflet.js as a global `L`.
declare const L: any;

export function polylinesFromItinerary(it: Itinerary): Array<Array<[number, number]>> {
  if (!it?.legs) return [];
  return it.legs
    .filter((l: any) => Array.isArray(l.polyline) && l.polyline.length >= 2)
    .map((l: any) => l.polyline);
}

export type MapHandle = {
  addPath: (p: Path) => void;
  setActive: (id: string | null) => void;
  clear: () => void;
  fitToPaths: () => void;
};

export function initMap(elId: string): MapHandle {
  const map = L.map(elId).setView([-37.81, 144.96], 11);
  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    maxZoom: 19, attribution: '© OpenStreetMap',
  }).addTo(map);

  const layers = new Map<string, any[]>();

  return {
    addPath(p: Path) {
      const lines = polylinesFromItinerary(p.itinerary);
      const polys = lines.map((coords) =>
        L.polyline(coords, { color: p.color, weight: 3 }).addTo(map),
      );
      polys.forEach((poly) =>
        poly.on('click', () => {
          document.dispatchEvent(new CustomEvent('chat:set-active', { detail: p.id }));
        }),
      );
      layers.set(p.id, polys);
    },
    setActive(id: string | null) {
      for (const [pid, polys] of layers) {
        const active = pid === id;
        polys.forEach((p) => p.setStyle({
          weight: active ? 5 : (id ? 2 : 3),
          opacity: id && !active ? 0.4 : 1,
        }));
      }
    },
    clear() {
      for (const [, polys] of layers) polys.forEach((p) => map.removeLayer(p));
      layers.clear();
    },
    fitToPaths() {
      const all: any[] = [];
      for (const [, polys] of layers) all.push(...polys);
      if (all.length === 0) return;
      const group = L.featureGroup(all);
      map.fitBounds(group.getBounds(), { padding: [40, 40] });
    },
  };
}
```

```ts
  dispatch({ type: 'set_active', id: e.detail });
});

async function tryGeolocate(): Promise<{ lat: number; lon: number } | undefined> {
  if (!navigator.geolocation) return undefined;
  try {
    return await new Promise((resolve, reject) => {
      navigator.geolocation.getCurrentPosition(
        (pos) => resolve({ lat: pos.coords.latitude, lon: pos.coords.longitude }),
        reject,
        { timeout: 1500, maximumAge: 60_000 },
      );
    });
  } catch {
    return undefined;
  }
}

render();
```
